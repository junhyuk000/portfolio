# 범위 직접 지정
plt.figure(figsize=(10, 5))  # 단위: 인치
x = [1, 2, 3, 4]
y = [28, 25, 26, 21]

plt.plot(x,y)
plt.xlim(0, 5)                    # x축의 범위를 0~5로 설정한다.
plt.ylim(10, 30)                  # y축의 범위를 10~30으로 설정한다.
# plt.xlim(left=0, right=5)       # x축의 범위는 왼쪽 0에서 오른쪽 5로 설정한다.
# plt.ylim(bottom=10, top=30)     # y축의 범위는 아래쪽 10에서 위쪽 30으로 설정한다.
plt.axis([0, 5, 10, 30])          # [xmin, xmax, ymin, ymax]
plt.show()

## 눈금설정
plt.figure(figsize=(10, 5))  # 단위: 인치
plt.plot([78, 75, 76, 72, 87, 80, 77])
# plt.xticks([0, 1, 2, 3], labels=['1달', '2달', '3달', '4달'], fontsize=12, color='r', rotation=45)
plt.xticks([0, 1, 2, 3, 4, 5, 6], labels=['2024-01-01', '2024-01-02', '2024-01-03', '2024-01-04', '2024-01-05', '2024-01-06', '2024-01-07'], fontsize=12, color='b', rotation = 45)
plt.yticks(range(70, 90, 2))
plt.show()
- 그래프의 크기를 가로10인치, 세로 5인치로 설정한다. - '성명'과 '급여' 컬럼을 각각 x축과 y축 데이터로 사용한다. - 원 모양의 마커를 표시하고, 선 스타일은 실선으로 지정하고, 선과 마커의 색상은 파랑으로 한다. - 제목은 '선수별 급여'로 지정하고, x축 제목은 '성명'으로 지정하고, y축 제목은 '급여(원)'으로 지정한다. - 그래프는 데이터 포인터의 위치를 쉽게 확인할 수 있도록 그리드를 표시한다.

import matplotlib.pyplot as plt

data = {
    '성명':['홍길동','김철수','이영희','박민수','정하나'],
    '급여':[3000000, 4500000, 2500000, 5000000, 3500000]
}

plt.figure(figsize=(10, 5))
plt.plot(data['성명'], data['급여'], marker='o', linestyle=':', color='blue')
plt.title('선수별 급여')
plt.xlabel('성명')
plt.ylabel('급여(원)')
plt.grid()
plt.show()

.bar()¶

  • 데이터 값을 수직 막대 그래프로 표시
In [65]:
import matplotlib.pyplot as plt

x = [1, 2, 3]
values = [100, 700, 500]
# x: 막대 그래프의 x축에 표시될 범주형 변수, values: 각 막대 그래프의 크기
# width : 너비 비율(0 ~ 1)
# edgecolor : 막대 테두리 
# linewidth : 막대 테두리 두께
# align : 'edge' 눈금 왼쪽, 'center' 눈금 가운데
# tick_label : 눈금 라벨 명
plt.bar(x, values, color='#03c75a', width=0.8, edgecolor='black', linewidth=5, align='edge', tick_label=['2021년','2022년','2023년'])
# plt.xticks(x, ['2021년','2022년','2023년'])
plt.show()
