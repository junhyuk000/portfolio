def get_news_titles(keyword, num_pages):

    driver = webdriver.Chrome()  # 옵션 사용 안할시 이문장만 있으면됨

    titles = []
    
    for page in range(1, num_pages + 1):
        url = f"https://search.naver.com/search.naver?ssc=tab.news.all&where=news&sm=tab_jum&query={keyword}&start={(page-1)*10 + 1}" # page에 10개씩뿌려지므로
        driver.get(url)

        time.sleep(1)

        soup = BeautifulSoup(driver.page_source,'html.parser')
        main_titles = soup.select('a.news_tit')
        for title in main_titles:
            titles.append({
                #'title': title.get('title'),
                #'link': title.get('href')
                'title': title['title'],
                'link': title['href']
            })

    print(f"총 {len(titles)}개의 기사 수집\n")

    for idx, item in enumerate(titles, 1):
        print(f"{idx}. {item['title']}")
        print(f"    링크: {item['link']}")
        print()

    driver.quit()

class NaverNewsCollector:
    def __init__(self):
        self.driver = webdriver.Chrome()
        
    def collect_news(self, keyword, num_pages):
        articles = []
        
        try:
            for page in range(1, num_pages + 1):
                url = f"https://search.naver.com/search.naver?where=news&query={keyword}&start={(page-1)*10 + 1}"
                self.driver.get(url)
                time.sleep(2)
                
                soup = BeautifulSoup(self.driver.page_source, 'html.parser')
                news_items = soup.select("div.news_wrap.api_ani_send")
                
                for item in news_items:
                    try:
                        title_element = item.select_one("a.news_tit")
                        if not title_element:
                            continue
                            
                        title = title_element['title']
                        link = title_element['href']
                        
                        press = item.select_one("a.press")
                        press_name = press.text if press else "알 수 없음"
                        
                        date = item.select_one("span.info")
                        date_text = date.text if date else "날짜 없음"
                        
                        articles.append({
                            '수집일시': datetime.now().strftime('%Y-%m-%d %H:%M:%S'),
                            '검색어': keyword,
                            '제목': title,
                            '언론사': press_name,
                            '보도일자': date_text,
                            '링크': link
                        })
                        
                    except Exception as e:
                        print(f"기사 추출 중 에러: {str(e)}")
                        continue
                
                print(f"{page}페이지 완료")
                
            return articles
            
        except Exception as e:
            print(f"에러 발생: {e}")
            return articles

    def save_to_csv(self, articles, keyword):
        try:
            df = pd.DataFrame(articles)
            
            output_dir = 'news_data'
            if not os.path.exists(output_dir):
                os.makedirs(output_dir)
            
            timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
            filename = f"{output_dir}/naver_news_{keyword}_{timestamp}.csv"
            
            df.to_csv(filename, index=False, encoding='utf-8-sig')
            
            print("\n=== 저장 완료 ===")
            print(f"파일 저장 위치: {filename}")
            print(f"총 기사 수: {len(articles)}개")
            print("\n=== 데이터 통계 ===")
            print("언론사별 기사 수:")
            print(df['언론사'].value_counts().head())
            
            return filename
            
        except Exception as e:
            print(f"CSV 저장 중 에러 발생: {str(e)}")
            return None

    def close(self):
        self.driver.quit()

def main():
    keyword = input("검색할 키워드를 입력하세요: ")
    num_pages = int(input("수집할 페이지 수를 입력하세요: "))
    
    collector = NaverNewsCollector()
    try:
        print(f"'{keyword}' 관련 뉴스 수집을 시작합니다...")
        articles = collector.collect_news(keyword, num_pages)
        
        if articles:
            csv_file = collector.save_to_csv(articles, keyword)
            print("\n=== 작업 완료 ===")
            if csv_file:
                print(f"CSV 파일: {csv_file}")
           
    finally:
        collector.close()

if __name__ == "__main__":
    main()
